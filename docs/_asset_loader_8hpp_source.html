50 std::optional<Handle<T> > Load(const std::filesystem::path &filePath) {
51 std::string pathStr = filePath.string();
52
53 if (m_AssetIds.contains(pathStr)) {
54 AssetId id = m_AssetIds.at(pathStr);
55 TypeId typeId = m_Assets.at(id).value().typeId;
56
57 if (typeId != GetTypeId<T>()) {
58 return std::nullopt;
59 }
60
61 return Handle<T>{.id = id, .typeId = typeId};
62 }
63
64 auto asset = Loader<T>::Load(*this, filePath, std::nullopt);
65 if (!asset.has_value()) {
66 return std::nullopt;
67 }
68
69 std::shared_ptr<T> ptr = std::make_shared<T>(std::move(asset.value()));
70
71 AssetId id = 0;
72 TypeId typeId = 0;
73
74 if (!m_FreeAssetIds.empty()) {
75 id = m_FreeAssetIds.back();
76 typeId = GetTypeId<T>();
77
78 m_FreeAssetIds.pop_back();
79
80 m_Assets[id] = AssetSlot{.data = ptr, .typeId = typeId};
81 } else {
82 id = m_Assets.size();
83 typeId = GetTypeId<T>();
84
85 m_Assets.emplace_back(AssetSlot{.data = ptr, .typeId = typeId});
86 }
87
88 m_AssetIds[pathStr] = id;
89 return Handle<T>{.id = id, .typeId = typeId};

168 template<typename T>
169 bool Unload(Handle<T> handle) {
170 if (handle.id >= m_Assets.size()) {
171 return false;
172 }
173
174 if (!m_Assets.at(handle.id).has_value()) {
175 Debug::LogWrn("Trying to unload an asset that is already unloaded!");
176 return false;
177 }
178
179 if (GetTypeId<T>() != m_Assets[handle.id].value().typeId) {
180 return false;
181 }
182
183 m_Assets[handle.id] = std::nullopt;
184
185 for (auto &[path, id]: m_AssetIds) {
186 if (id == handle.id) {
187 m_AssetIds.erase(path);
188 break;
189 }
190 }
191
192 m_FreeAssetIds.push_back(handle.id);
193
194 return true;
195 }

294 outputMaterials.reserve(materials.size());
295 for (auto &[color, metallic, roughness, colorMapPath, metallicMapPath, roughnessMapPath]: materials) {
296 Material material;

301 material.colorMap = colorMap;
302 }
303
304 if (metallicMapPath) {
305 const auto metallicMap = loader.Load<Texture2D>(metallicMapPath.value(), config);
306 material.metallicMap = metallicMap;
307 }
308
309 if (roughnessMapPath) {
310 const auto roughnessMap = loader.Load<Texture2D>(roughnessMapPath.value(), config);
311 material.roughnessMap = roughnessMap;
312 }
313
314 material.color = color;
315 material.metallic = metallic;
316 material.roughness = roughness;
317
318 outputMaterials.push_back(material);
319 }

322 for (auto &[data, materialIndex]: meshes) {
323 model.meshes.push_back(std::move(Mesh::Create(data).value()));
